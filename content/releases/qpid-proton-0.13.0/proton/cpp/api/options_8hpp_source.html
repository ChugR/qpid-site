64  int parse() {
65  int arg = 1;
66  for (; arg < argc_ && argv_[arg][0] == '-'; ++arg) {
67  opts::iterator i = opts_.begin();
68  while (i != opts_.end() && !(*i)->parse(argc_, argv_, arg))
69  ++i;
70  if (i == opts_.end())
71  throw bad_option(std::string("unknown option ") + argv_[arg]);
72  }
73  if (help_) throw bad_option("");
74  return arg;
75  }

116  bool parse(int argc, char const * const * argv, int &i) {
117  std::string arg(argv[i]);
118  if (arg == short_ || arg == long_) {
119  if (i < argc-1) {
120  set_value(arg, argv[++i]);
121  return true;
122  } else {
123  throw bad_option("missing value for " + arg);
124  }
125  }
126  if (arg.compare(0, long_.size(), long_) == 0 && arg[long_.size()] == '=' ) {
127  set_value(long_, arg.substr(long_.size()+1));
128  return true;
129  }
130  return false;
131  }
